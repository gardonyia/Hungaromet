
import streamlit as st
import requests
import zipfile
import io
import pandas as pd
from datetime import datetime

st.title("Magyarországi napi hőmérsékleti szélsőértékek")
st.write("Adatok forrása: [Hungaromet ODP](https://odp.met.hu/weather/weather_reports/s):
    today = datetime.now().strftime("%Y%m%d")
    url = f"https://odp.met.hu/weather/weather_reports/synoptic/hungary/daily/csv/HABP_1D_{today}.zip"

    st.write(f"Adatok letöltése: {url}")
    response = requests.get(url)

    if response.status_code != 200:
        st.error("Nem sikerült letölteni az adatokat. Lehet, hogy még nincs adat a mai napra.")
    else:
        with zipfile.ZipFile(io.BytesIO(response.content)) as z:
            csv_name = f"HABP_1D_{today}.csv"
            with z.open(csv_name) as csv_file:
                df = pd.read_csv(csv_file, sep=';', encoding='utf-8')

        if 'tn' not in df.columns or 'tx' not in df.columns:
            st.error("A fájl nem tartalmaz tn vagy tx oszlopot!")
        else:
            min_temp = df['tn'].min()
            max_temp = df['tx'].max()
            min_station = df.loc[df['tn'].idxmin(), 'stationName']
            max_station = df.loc[df['tx'].idxmax(), 'stationName']
            date_str = datetime.now().strftime("%Y.%m.%d")

            st.success(f"A hőmérsékleti szélső értékek {date_str}-re vonatkozóan:")
            st.write(f"**Maximum:** {max_temp} °C (állomás: {max_station})")
            st.write(f"**Minimum:** {min_temp} °C (állomás: {min_station})")
